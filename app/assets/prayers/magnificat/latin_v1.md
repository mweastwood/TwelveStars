{Magnificat anima mea Dominum,|magnificat_1}
{et exsultavit spiritus meus in Deo salvatore meo,|magnificat_2}
{quia respexit humilitatem ancillae suae.|magnificat_3}
{Ecce enim ex hoc beatam me dicent omnes generationes,|magnificat_4}
{quia fecit mihi magna, qui potens est,|magnificat_5}
{et sanctum nomen eius,|magnificat_6}
{et misericordia eius in progenies et progenies|magnificat_7}
{timentibus eum.|magnificat_8}
{Fecit potentiam in brachio suo,|magnificat_9}
{dispersit superbos mente cordis sui;|magnificat_10}
{deposuit potentes de sede|magnificat_11}
{et exaltavit humiles;|magnificat_12}
{esurientes implevit bonis|magnificat_13}
{et divites dimisit inanes.|magnificat_14}
{Suscepit Israel puerum suum,|magnificat_15}
{recordatus misericordiae,|magnificat_16}
{sicut locutus est ad patres nostros,|magnificat_17}
{Abraham et semini eius in saecula.|magnificat_18}
